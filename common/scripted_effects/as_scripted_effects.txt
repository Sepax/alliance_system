sort_highest_dev_setup = {
	every_country = {
		export_to_variable = {
			which = dev
			value = total_development
		}
		save_event_target_as = target
		every_subject_country = {
			export_to_variable = {
				which = dev
				value = total_development
			}
			set_variable = {
				which = temp
				which = dev
			}
			divide_variable = {
				which = temp
				value = 2
			}
			event_target:target = {
				set_variable = {
					which = temp
					which = PREV
				}
				change_variable = {
					which = dev
					which = temp
				}
			}
		}
		flatten = {
			float = dev
		}
	}
}

sort_highest_dev_cleanup = {
	every_country = {
		clr_country_flag = top_dev_sorted
	}
}

sort_highest_dev = {
	#target
	random_country = {
		limit = {
			NOT = {
				has_country_flag = top_dev_sorted
			}
		}
		save_global_event_target_as = $target$
	}
	every_country = {
		limit = {
			NOT = {
				has_country_flag = top_dev_sorted
			}
			check_variable = {
				which = dev
				which = event_target:$target$
			}
		}
		save_global_event_target_as = $target$
	}
	event_target:$target$ = {
		set_country_flag = top_dev_sorted
		set_variable = {
			which = global_top_dev
			which = dev
		}
	}
}

# Checks the size of the root country and sets the alliance cost of said country
set_alliance_cost = {
	#if = {
	# limit = {
	# 	ai = no
	# }
	hidden_effect = {
		alliance_system_config = yes
		set_variable = {
			which = alliance_cost
			value = 0
		}
		set_variable = {
			which = alliance_points
			which = base_alliance_points
		}
	}
	if = {
		limit = {
			check_variable = {
				which = dev
				which = global_top_dev
			}
		}
		change_variable = {
			which = alliance_cost
			which = colossal_points
		}
	}
	else_if = {
		limit = {
			check_variable = {
				which = dev
				which = major_dev
			}
		}
		change_variable = {
			which = alliance_cost
			which = major_points
		}
	}
	else_if = {
		limit = {
			check_variable = {
				which = dev
				which = prominent_dev
			}
		}
		change_variable = {
			which = alliance_cost
			which = prominent_points
		}
	}
	else_if = {
		limit = {
			check_variable = {
				which = dev
				which = notable_dev
			}
		}
		change_variable = {
			which = alliance_cost
			which = notable_points
		}
	}
	else_if = {
		limit = {
			check_variable = {
				which = dev
				which = minor_dev
			}
		}
		change_variable = {
			which = alliance_cost
			which = minor_points
		}
	}
	else = {
		change_variable = {
			which = alliance_cost
			which = puny_points
		}
	}
	#}
}

alliance_scanner = {
	set_alliance_cost = yes
	subtract_variable = {
		which = alliance_points
		which = alliance_cost
	}
	save_event_target_as = target
	every_ally = {
		# limit = {
		# 	ai = no
		# }
		set_variable = {
			which = temp
			which = alliance_cost
		}
		event_target:target = {
			set_variable = {
				which = temp
				which = alliance_points
			}
			subtract_variable = {
				which = temp
				which = PREV
			}
			set_variable = {
				which = alliance_points
				which = temp
			}
		}
	}
	if = {
		limit = {
			NOT = {
				check_variable = {
					which = alliance_points
					value = 0
				}
			}
		}
		set_country_flag = too_many_alliances
	}
	else = {
		clr_country_flag = too_many_alliances
	}
}

flatten = {

	}
	multiply_variable = {
		which = $float$
		value = 1000
	}
}
