is_player = {
    ai = no
}

is_player_allied_with_player = {
    is_player = yes
    alliance_with = $who$
    $who$ = {
        is_player = yes
    }
}

is_player_subject_of_player = {
    $who$ = {
        is_player = yes
    }
    is_player = yes
    is_subject_of = $who$
    NOT = {
        is_subject_of_type = tributary_state
    }
}

is_free_player_subject_of_player = {
    is_player_subject_of_player = { who = $who$ }
    current_age = age_of_discovery
    OR = {
        tag = SWE
        tag = NOR
        tag = HOL
        tag = FLA
        tag = BRB
        tag = LIT
    }
}

is_player_with_player_tributary = {
    is_player = yes
    $who$ = {
        is_player = yes
        is_subject_of = PREV
        is_subject_of_type = tributary_state
    }
}

is_player_guaranteeing_player = {
    is_player = yes
    $who$ = {
        is_player = yes
        guaranteed_by = PREV
    }
}

is_player_supporting_player = {
    is_player = yes
    is_supporting_independence_of = $who$
    $who$ = {
        is_player = yes
    }
}

in_treaty_with = {
    OR = {
        has_country_flag = allied_with_@$who$
        has_country_flag = subject_of_@$who$
        has_country_flag = has_tributary_@$who$
        has_country_flag = guaranteeing_@$who$
        has_country_flag = supporting_@$who$
    }
}

would_break_rules = {
    OR = {
        AND = {
            $compare$ = { is_variable_equal = { which = as_cost which = as_cost_great_power } }
            NOT = { check_variable = { which = as_points which = as_cost_great_power } } 
        }
        AND = {
            $compare$ = { is_variable_equal = { which = as_cost which = as_cost_major_power } }
            NOT = { check_variable = { which = as_points which = as_cost_major_power } } 
        }
        AND = {
            $compare$ = { is_variable_equal = { which = as_cost which = as_cost_prom_power } }
            NOT = { check_variable = { which = as_points which = as_cost_prom_power } } 
        }
        AND = {
            $compare$ = { is_variable_equal = { which = as_cost which = as_cost_notable_power } }
            NOT = { check_variable = { which = as_points which = as_cost_notable_power } } 
        }
        AND = {
            $compare$ = { is_variable_equal = { which = as_cost which = as_cost_minor_power } }
            NOT = { check_variable = { which = as_points which = as_cost_minor_power } } 
        }
        AND = {
            $compare$ = { is_variable_equal = { which = as_cost which = as_cost_puny_power } }
            NOT = { check_variable = { which = as_points which = as_cost_puny_power } } 
        }
    }
}
