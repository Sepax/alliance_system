custom_button = {
    name = as_show_window
    potential = {
        ai = no
    }
    effect = {
        if = {
            limit = {
                has_country_flag = as_ui_open 
            }
            clr_country_flag = as_ui_open 
        }
        else {
            set_country_flag = as_ui_open
            update_treaties = yes
        }
    }
    tooltip = as_toggle_ui_button
}
